        function loadLog() {
            const entries = JSON.parse(localStorage.getItem('quintonLog') || '[]');
            const container = document.getElementById('logEntries');
            container.innerHTML = '';
            
            entries.forEach(entry => {
                const div = document.createElement('div');
                div.className = 'log-entry';
                div.innerHTML = <strong>\( {entry.time}</strong><br> \){entry.text};
                container.appendChild(div);
            });
        }

        function saveLogEntry() {
            const input = document.getElementById('logInput');
            const text = input.value.trim();
            if (!text) return;

            const entries = JSON.parse(localStorage.getItem('quintonLog') || '[]');
            const now = new Date();
            const timeStr = now.toLocaleString('en-CA', { 
                year: 'numeric', month: 'short', day: 'numeric',
                hour: '2-digit', minute: '2-digit' 
            });

            entries.unshift({ time: timeStr, text: text });
            if (entries.length > 30) entries.pop();

            localStorage.setItem('quintonLog', JSON.stringify(entries));
            input.value = '';
            loadLog();
        }

        function clearLog() {
            if (confirm('Clear all sighting logs?')) {
                localStorage.removeItem('quintonLog');
                loadLog();
            }
        }

        window.onload = () => {
            loadLog();
            
            document.getElementById('logInput').addEventListener('keypress', function(e) {
                if (e.key === 'Enter' && !e.shiftKey) {
                    e.preventDefault();
                    saveLogEntry();
                }
            });
        };
